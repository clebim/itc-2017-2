#include <conio.h>
#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

int main()
{
	int x = 1, y = 0;
	setlocale(LC_ALL, "portuguese");
	
	while(y <= 0 || y%2==0)
	{
		system("cls");
		printf("Digite um numero impar e positivo: ");
		scanf("%d", &y);
	}
	printf("\nNumeros impares no intervalo: \n\n");
	while(y >= x)
	{
		printf("%d ", y);
		y = y - 2;
	}
	getch();
	return 0;
}
